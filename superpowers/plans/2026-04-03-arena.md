# Arena Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `arena`, a new frontend project bootstrapped from `starter`, that can define sandbox benchmark suites, run provider-backed benchmark executions through a shared runner contract, and present leaderboard, compare, and run detail views for the first four sandbox providers.

**Architecture:** `arena` should stay platform-first. Shared benchmark types, suite definitions, result normalization, and route loaders live in app-level domain files. Provider-specific differences stay in runner adapters and provider metadata. The first release uses fixture-backed runner implementations so the product shell, normalized result model, and comparison UI are complete before live sandbox integrations deepen.

**Tech Stack:** `starter` scaffold, TanStack Start, TanStack Router, React 19, Zustand, Zod, Tailwind CSS v4, Vite+, Vitest.

- [ ] **Step 7: Implement scorecard building and dimension sorting**

Create `arena/src/domain/benchmark/scorecard.ts`:

```ts
import type {
  BenchmarkRun,
  LeaderboardDimension,
  SandboxProvider,
  ScoreCard,
  TaskSuite,
} from "@/domain/benchmark/types";

function getNumericMetric(run: BenchmarkRun, key: string) {
  const metric = run.metrics.find((entry) => entry.key === key);
  return typeof metric?.value === "number" ? metric.value : 0;
}

export function buildScoreCard(run: BenchmarkRun): ScoreCard {
  return {
    providerId: run.providerId,
    runId: run.id,
    quality: {
      success: run.status === "passed",
      summary: run.summary,
    },
    executionMetrics: {
      durationMs: getNumericMetric(run, "durationMs"),
      setupMs: getNumericMetric(run, "setupMs"),
      installMs: getNumericMetric(run, "installMs"),
    },
    reviewData: {
      artifactCount: run.artifacts.length,
      hasCommandLog: run.artifacts.some((artifact) => artifact.key === "commands"),
    },
  };
}

function getDimensionValue(run: BenchmarkRun, dimension: LeaderboardDimension, suites: TaskSuite[]) {
  switch (dimension) {
    case "stability":
      return run.status === "passed" ? 1 : 0;
    case "speed":
      return -getNumericMetric(run, "durationMs");
    case "coverage": {
      const suite = suites.find((entry) => entry.id === run.suiteId);
      return suite ? suite.taskCases.length : 0;
    }
    case "cost":
      return 0;
    case "review":
      return run.artifacts.length;
  }
}

export function sortProvidersByDimension(args: {
  providers: SandboxProvider[];
  suites: TaskSuite[];
  runs: BenchmarkRun[];
  dimension: LeaderboardDimension;
}) {
  return args.providers
    .map((provider) => {
      const latestRun = args.runs.find((run) => run.providerId === provider.id);

      return {
        provider,
        run: latestRun,
        value: latestRun ? getDimensionValue(latestRun, args.dimension, args.suites) : -1,
      };
    })
    .sort((left, right) => right.value - left.value);
}
```

import type { BenchmarkTaskInput, SandboxRunnerAdapter } from "@/runners/contract";
import { boxliteRunner } from "@/runners/providers/boxlite";

import { rivetRunner } from "@/runners/providers/rivet";
import { spriteRunner } from "@/runners/providers/sprite";

const runnerRegistry = new Map<string, SandboxRunnerAdapter>([
  [e2bRunner.providerId, e2bRunner],
  [spriteRunner.providerId, spriteRunner],
  [boxliteRunner.providerId, boxliteRunner],
  [rivetRunner.providerId, rivetRunner],
]);

export async function runBenchmarkCase(task: BenchmarkTaskInput): Promise<BenchmarkRun> {
  const runner = runnerRegistry.get(task.providerId);

  if (!runner) {
    throw new Error(`Unknown provider: ${task.providerId}`);
  }

  return runner.run(task);
}

import { sandboxRunFixtures } from "@/domain/benchmark/fixtures";
import { buildScoreCard, sortProvidersByDimension } from "@/domain/benchmark/scorecard";
import type { LeaderboardDimension } from "@/domain/benchmark/types";
import { sandboxProviders } from "@/features/providers/providers";
import { sandboxSuites } from "@/features/sandbox/suites";

export function getArenaOverview() {
  return {
    trackId: "sandbox",
    providerCount: sandboxProviders.length,
    suiteCount: sandboxSuites.length,
    latestRunCount: sandboxRunFixtures.length,
    providers: sandboxProviders,
    runs: sandboxRunFixtures,
  };
}

export function getLeaderboardView(dimension: LeaderboardDimension) {
  const entries = sortProvidersByDimension({
    providers: sandboxProviders,
    suites: sandboxSuites,
    runs: sandboxRunFixtures,
    dimension,
  }).map((entry) => ({
    provider: entry.provider,
    run: entry.run,
    scoreCard: entry.run ? buildScoreCard(entry.run) : null,
  }));

  return { dimension, entries };
}

export function getCompareView(providerIds: string[]) {
  const providers = sandboxProviders.filter((provider) => providerIds.includes(provider.id));
  const runs = sandboxRunFixtures.filter((run) => providerIds.includes(run.providerId));

  return {
    providers,
    runs,
    rows: [
      {
        label: "Focus",
        values: providers.map((provider) => provider.focus),
      },
      {
        label: "Capabilities",
        values: providers.map((provider) => provider.capabilities.join(", ")),
      },
      {
        label: "Latest summary",
        values: providers.map(
          (provider) => runs.find((run) => run.providerId === provider.id)?.summary ?? "No runs",
        ),
      },
    ],
  };
}

export function getRunDetail(runId: string) {
  const run = sandboxRunFixtures.find((entry) => entry.id === runId);

  if (!run) {
    return null;
  }

  const provider = sandboxProviders.find((entry) => entry.id === run.providerId);
  const suite = sandboxSuites.find((entry) => entry.id === run.suiteId);
  const taskCase = suite?.taskCases.find((entry) => entry.id === run.taskCaseId);

  return {
    run,
    provider,
    suite,
    taskCase,
    scoreCard: buildScoreCard(run),
  };
}
