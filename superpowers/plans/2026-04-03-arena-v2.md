# Arena V2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Update `arena` so its first real release uses `OrbStack machine`, `BoxLite`, and `Sprite` as the default provider set, serves run and comparison data from a lightweight built-in backend, persists runs to local JSON files, and executes the first live benchmark path through `BoxLite`.

**Architecture:** Keep `arena` as one project, but move the source of truth for providers, runs, and leaderboard data into backend services under `src/server/arena`. The frontend should consume APIs and render page sections, while the backend owns provider registration, JSON persistence, run creation, and the first live provider adapter. Frontend cleanup should be medium-weight: reduce route/data coupling without rewriting the UI from scratch.

**Tech Stack:** TanStack Start, Nitro server routes, React 19, Tailwind CSS v4, Vite+, Vitest, Node.js filesystem APIs, Zod.

- [ ] **Step 6: Implement the JSON store**

Create `arena/src/server/arena/repository/json-store.ts`:

```ts
import { mkdir, readFile, writeFile } from "node:fs/promises";
import { join } from "node:path";

import type { BenchmarkRun } from "@/domain/benchmark/types";

async function readJsonFile<T>(filePath: string, fallback: T): Promise<T> {
  try {
    return JSON.parse(await readFile(filePath, "utf8")) as T;
  } catch {
    return fallback;
  }
}

export function createJsonArenaStore(rootDir: string) {
  const runsDir = join(rootDir, "runs");
  const indexPath = join(runsDir, "index.json");

  return {
    async listRuns() {
      return readJsonFile<BenchmarkRun[]>(indexPath, []);
    },
    async writeRun(run: BenchmarkRun) {
      await mkdir(runsDir, { recursive: true });

      const existingRuns = await readJsonFile<BenchmarkRun[]>(indexPath, []);
      const nextRuns = [run, ...existingRuns.filter((entry) => entry.id !== run.id)];

      await writeFile(indexPath, JSON.stringify(nextRuns, null, 2));
      await writeFile(join(runsDir, `${run.id}.json`), JSON.stringify(run, null, 2));
    },
    async getRun(runId: string) {
      return readJsonFile<BenchmarkRun | null>(join(runsDir, `${runId}.json`), null);
    },
  };
}
```

- [ ] **Step 3: Implement run and leaderboard services**

Create `arena/src/server/arena/service/run-service.ts`:

```ts
import { randomUUID } from "node:crypto";
import { join } from "node:path";

import { buildScoreCard, sortProvidersByDimension } from "@/domain/benchmark/scorecard";
import type { BenchmarkRun, LeaderboardDimension } from "@/domain/benchmark/types";
import { arenaProviders } from "@/server/arena/providers";
import { createJsonArenaStore } from "@/server/arena/repository/json-store";
import { arenaSeedRuns } from "@/server/arena/fixtures";
import { sandboxSuites } from "@/features/sandbox/suites";

export function createRunService(projectRoot: string) {
  const store = createJsonArenaStore(join(projectRoot, ".local"));

  return {
    async listRuns() {
      const persistedRuns = await store.listRuns();
      return persistedRuns.length > 0 ? persistedRuns : arenaSeedRuns;
    },
    async getRun(runId: string) {
      const persistedRun = await store.getRun(runId);
      return persistedRun ?? arenaSeedRuns.find((run) => run.id === runId) ?? null;
    },
    async createFixtureRun(input: {
      providerId: string;
      suiteId: BenchmarkRun["suiteId"];
      taskCaseId: string;
    }) {
      const template = arenaSeedRuns.find(
        (run) =>
          run.providerId === input.providerId &&
          run.suiteId === input.suiteId &&
          run.taskCaseId === input.taskCaseId,
      );

      if (!template) {
        throw new Error(`Missing seed run for provider: ${input.providerId}`);
      }

      const run = {
        ...structuredClone(template),
        id: `${template.id}-${randomUUID()}`,
      } satisfies BenchmarkRun;

      await store.writeRun(run);
      return run;
    },
  };
}

export function createLeaderboardService(projectRoot: string) {
  const runService = createRunService(projectRoot);

  return {
    async getLeaderboard(dimension: LeaderboardDimension) {
      const runs = await runService.listRuns();
      return sortProvidersByDimension({
        providers: arenaProviders,
        suites: sandboxSuites,
        runs,
        dimension,
      }).map((entry) => ({
        provider: entry.provider,
        run: entry.run,
        scoreCard: entry.run ? buildScoreCard(entry.run) : null,
      }));
    },
  };
}
```

export async function runBoxliteTask(input: {

  return {
    ...structuredClone(fixture),
    id: `${fixture.id}-${randomUUID()}`,
    source: "live",
    summary: `BoxLite live run: ${fixture.summary}`,
  };
}

- [ ] **Step 3: Add the frontend API client**

Create `arena/src/lib/api/arena.ts`:

```ts
import { ofetch } from "ofetch";

import { getApiBaseUrl } from "@/lib/api-base-url";

const arenaApi = ofetch.create({ baseURL: getApiBaseUrl().replace(/\/api\/v1$/, "") });

export async function fetchArenaProviders() {
  return arenaApi<{ providers: unknown[] }>("/api/providers");
}

export async function fetchArenaRuns() {
  return arenaApi<{ runs: unknown[] }>("/api/runs");
}

export async function fetchArenaRun(runId: string) {
  return arenaApi<{ run: unknown }>(`/api/runs/${runId}`);
}

export async function fetchArenaLeaderboard(dimension: string) {
  return arenaApi<{ dimension: string; entries: unknown[] }>("/api/leaderboard", {
    query: { dimension },
  });
}

export async function fetchArenaCompare(providerIds: string[]) {
  return arenaApi<{ providers: unknown[]; runs: unknown[] }>("/api/compare", {
    query: { provider: providerIds },
  });
}
```
